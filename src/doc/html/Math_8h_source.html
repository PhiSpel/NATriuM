00060 // Asserting abs[ <a,b> / (|a| |b|) ] > 0.99 is equivalent to:    angle(a,b) < 8 degrees.
00061 inline bool is_angle_small(dealii::Tensor<1, 2> vector1,
00062                 dealii::Tensor<1, 2> vector2, double thresholdDegrees = 0.0) {
